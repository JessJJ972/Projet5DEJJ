services:
  mongo:
    image: mongo:7
    container_name: mongo
    restart: unless-stopped
    environment:
      MONGO_INITDB_ROOT_USERNAME: root
      MONGO_INITDB_ROOT_PASSWORD: projetde
      MONGO_INITDB_DATABASE: healthcare
    ports:
      - "27017:27017"
    volumes:
      - mongo_data:/data/db

  migration:
    image: mongo:7
    container_name: mongo-csv-import
    depends_on:
      - mongo
    volumes:
      - ./data:/data:ro
    command: >
      mongoimport
      --host mongo
      -u root
      -p example
      --authenticationDatabase admin
      --db healthcare
      --collection patients
      --type csv
      --headerline
      --drop
      --file /data/healthcare_dataset.csv

  mongo_data:
